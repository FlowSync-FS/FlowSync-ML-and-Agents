-- =============================================================================
-- FlowSync — Database Views
-- Run AFTER schema.sql. These views are queried by orchestrator.py
-- to build agent state — they add product_name, ptr, remaining_qty, etc.
-- that the raw ML output tables don't store (keeping those tables narrow).
-- =============================================================================


-- -----------------------------------------------------------------------------
-- v_expiry_predictions
-- Enriches expiry_predictions with remaining_qty, ptr, and product_name.
-- Queried by orchestrator._build_agent_state() for ExpiryPreventionAgent.
-- -----------------------------------------------------------------------------
CREATE OR REPLACE VIEW v_expiry_predictions AS
SELECT
    ep.id,
    ep.depot_id,
    ep.batch_id,
    ep.run_date,
    ep.expiry_risk_score,
    ep.recommended_liquidation_date,
    ep.method,
    ep.created_at,
    b.product_id,
    b.quantity_remaining                AS remaining_qty,
    p.ptr,
    p.canonical_name                    AS product_name
FROM expiry_predictions ep
JOIN batches  b ON b.id  = ep.batch_id
JOIN products p ON p.id  = b.product_id;


-- -----------------------------------------------------------------------------
-- v_stockout_risks
-- Renames stockout_calculations, adds product_name and predicted_units_14d.
-- Queried by orchestrator._build_agent_state() for ReorderAgent.
-- Note: the orchestrator historically referenced this as "stockout_predictions"
-- (wrong name) — the view bridges that while keeping the canonical table name.
-- -----------------------------------------------------------------------------
CREATE OR REPLACE VIEW v_stockout_risks AS
SELECT
    sc.id,
    sc.depot_id,
    sc.product_id,
    sc.run_date,
    sc.current_stock,
    sc.days_until_stockout,
    sc.lead_time_days,
    sc.will_stockout,
    sc.created_at,
    p.canonical_name                    AS product_name,
    dp.predicted_units_14d
FROM stockout_calculations sc
JOIN products p ON p.id = sc.product_id
LEFT JOIN demand_predictions dp
    ON  dp.product_id = sc.product_id
    AND dp.depot_id   = sc.depot_id
    AND dp.run_date   = sc.run_date;


-- -----------------------------------------------------------------------------
-- v_fefo_rankings
-- Enriches fefo_rankings with units_available, expiry_date, product_name,
-- and predicted_units_14d from demand_predictions on the same run_date.
-- Queried by orchestrator._build_agent_state() for DispatchAgent.
-- -----------------------------------------------------------------------------
CREATE OR REPLACE VIEW v_fefo_rankings AS
SELECT
    fr.id,
    fr.depot_id,
    fr.batch_id,
    fr.run_date,
    fr.priority_rank,
    fr.priority_score,
    fr.ml_override,
    fr.expiry_risk_score,
    fr.days_till_expiry,
    fr.created_at,
    b.product_id,
    b.quantity_remaining                AS units_available,
    b.expiry_date,
    p.canonical_name                    AS product_name,
    dp.predicted_units_14d
FROM fefo_rankings fr
JOIN batches  b  ON b.id  = fr.batch_id
JOIN products p  ON p.id  = b.product_id
LEFT JOIN demand_predictions dp
    ON  dp.product_id = b.product_id
    AND dp.depot_id   = fr.depot_id
    AND dp.run_date   = fr.run_date;


-- -----------------------------------------------------------------------------
-- v_anomaly_flags
-- Enriches anomaly_flags with product_name via products join.
-- Falls back to the stored product_name column when product_id is NULL.
-- Queried by orchestrator._build_agent_state() for AnomalyAgent.
-- -----------------------------------------------------------------------------
CREATE OR REPLACE VIEW v_anomaly_flags AS
SELECT
    af.id,
    af.depot_id,
    af.movement_id,
    af.batch_id,
    af.product_id,
    af.run_date,
    af.z_score,
    af.action,
    af.quantity,
    af.movement_type,
    af.created_at,
    COALESCE(p.canonical_name, af.product_name) AS product_name
FROM anomaly_flags af
LEFT JOIN products p ON p.id = af.product_id;
